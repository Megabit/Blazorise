<Button Color="Color.Primary" Clicked="@ShowOffcanvas">Show Offcanvas</Button>

<Offcanvas @ref="offcanvasRef">
    <OffcanvasBody>
        <Field>
            <FieldLabel>Name</FieldLabel>
            <TextEdit Placeholder="Enter name..." />
        </Field>
        <Field>
            <FieldLabel>Surname</FieldLabel>
            <TextEdit Placeholder="Enter surname..." />
        </Field>
        <div class="d-flex justify-content-end mt-4">
            <Button Color="Color.Secondary" Clicked="@HideOffcanvas">Close</Button>
            <Button Color="Color.Primary" Clicked="@HideOffcanvas">Save Changes</Button>
        </div>
    </OffcanvasBody>
</Offcanvas>

@code {
    // reference to the offcanvas component
    private Offcanvas offcanvasRef;

    private Task ShowOffcanvas()
    {
        return offcanvasRef.Show();
    }

    private Task HideOffcanvas()
    {
        return offcanvasRef.Hide();
    }
}
